<body>
    <h1>Pesquise um Assunto e Faça um Quiz</h1>
    <div id="search">
        <input type="text" id="topic" placeholder="Digite um assunto (ex: Brasil)">
        <button onclick="searchTopic()">Pesquisar</button>
    </div>
    <div id="quiz">
        <h2>Quiz sobre: <span id="topicTitle"></span></h2>
        <div id="questions"></div>
        <button onclick="checkAnswers()">Verificar Respostas</button>
        <div id="results"></div>
    </div>

    <script>
        let quizData = [];

        async function searchTopic() {
            const topic = document.getElementById('topic').value.trim();
            if (!topic) {
                alert('Digite um assunto!');
                return;
            }

            try {
                const response = await fetch(`https://pt.wikipedia.org/api/rest_v1/page/summary/${encodeURIComponent(topic)}`);
                if (!response.ok) {
                    alert('Assunto não encontrado na Wikipedia!');
                    return;
                }
                const data = await response.json();
                const extract = data.extract;

                // Gerar quiz baseado no resumo
                generateQuiz(topic, extract);
                document.getElementById('topicTitle').textContent = topic;
                document.getElementById('search').style.display = 'none';
                document.getElementById('quiz').style.display = 'block';
            } catch (error) {
                console.error('Erro ao buscar dados:', error);
                alert('Erro ao conectar à Wikipedia.');
            }
        }

        function generateQuiz(topic, extract) {
            // Dividir o resumo em sentenças
            const sentences = extract.split('. ').filter(s => s.length > 10);
            quizData = [];

            // Gerar 10 perguntas de múltipla escolha simples baseadas nas sentenças
            for (let i = 0; i < Math.min(10, sentences.length); i++) {
                const sentence = sentences[i];
                // Criar pergunta: "O que é mencionado sobre [palavra-chave]?"
                const words = sentence.split(' ');
                const keyWord = words[Math.floor(words.length / 2)] || topic;
                const question = `O que é mencionado sobre "${keyWord}" na seguinte frase: "${sentence}."?`;

                // Opções: uma correta (parte da frase), outras incorretas
                const correctAnswer = sentence.substring(0, 50) + '...'; // Simplificado
                const options = [
                    correctAnswer,
                    'Não é mencionado.',
                    'É um fato histórico.',
                    'Relacionado a ciência.'
                ];
                // Embaralhar opções
                options.sort(() => Math.random() - 0.5);

                quizData.push({
                    question,
                    options,
                    correct: correctAnswer
                });
            }

            displayQuiz();
        }

        function displayQuiz() {
            const questionsDiv = document.getElementById('questions');
            questionsDiv.innerHTML = '';
            quizData.forEach((q, index) => {
                const qDiv = document.createElement('div');
                qDiv.className = 'question';
                qDiv.innerHTML = `<p>${index + 1}. ${q.question}</p>`;
                q.options.forEach(option => {
                    const label = document.createElement('label');
                    label.innerHTML = `<input type="radio" name="q${index}" value="${option}"> ${option}<br>`;
                    qDiv.appendChild(label);
                });
                questionsDiv.appendChild(qDiv);
            });
        }

        function checkAnswers() {
            let score = 0;
            const resultsDiv = document.getElementById('results');
            resultsDiv.innerHTML = '<h3>Resultados:</h3>';
            quizData.forEach((q, index) => {
                const selected = document.querySelector(`input[name="q${index}"]:checked`);
                const userAnswer = selected ? selected.value : null;
                const isCorrect = userAnswer === q.correct;
                if (isCorrect) score++;
                resultsDiv.innerHTML += `<p>Pergunta ${index + 1}: ${isCorrect ? 'Correta' : 'Incorreta'} (Correta: ${q.correct})</p>`;
            });
            resultsDiv.innerHTML += `<p><strong>Pontuação: ${score}/10</strong></p>`;
        }
    </script>
